给定一个字符串`s`，请你找出其中不含有重复字符的最长**子串**的长度。

示例 1:

输入: s = "abcabcbb"
输出: 3
解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。
示例 2:

输入: s = "bbbbb"
输出: 1
解释: 因为无重复字符的最长子串是 "b"，所以其长度为 1。
示例 3:

输入: s = "pwwkew"
输出: 3
解释: 因为无重复字符的最长子串是 "wke"，所以其长度为 3。
请注意，你的答案必须是 子串 的长度，"pwke" 是一个子序列，不是子串。

```js
var lengthOfLongestSubstring = function (s) {
    const set = new Set();
    let maxLen = 0;
    // 默认视窗开始位置是0 也就是s的第一个字符
    let start = 0;
    // 如果需要返回子串
    let str = '';

    // 依次遍历字符串中的字符
    for (let i = 0; i < s.length; i++) {
        // 如果当前的字符在之前已经出现过 以下代码会将set中当前字符前面的字符全部删掉
        while (set.has(s[i])) {
            // 删除视窗start的字符
            set.delete(s[start]);
            // 视窗向右移动一位
            start++;
        }
        // 向set中添加当前字符
        set.add(s[i]);

        // maxLen是上一次不重复的字符长度 本次长度从start + 1开始 到i结束 长度为 i - start + 1
        // maxLen = Math.max(maxLen, i - start + 1 );

        // 如果需要返回子串
        if (i - start + 1 > maxLen) {
            // 与上面一样的意思
            maxLen = i - start + 1;
            // 从视窗左侧start到结束 slice不包含end项 所以+1
            str = s.slice(start, i + 1);
        }
    }
    return {maxLen, str};
};
```
